fun fact_recursive(n) {
    if n <= 1 {
        return 1
    }
    return n * fact_recursive(n-1)
}

assert(fact_recursive(9) == 362880) 
// 10 stackoverflows

fun fact_while(n) {
    let res = 1
    let i = 2
    while i <= n {
        res = res * i
        i = i + 1
    }
    return res
}

assert(fact_while(12) == 479001600) 
// 13 -> multiplication overflow
