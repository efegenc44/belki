fun fib(n) {
    if n <= 2 {
        return 1
    }
    return fib(n-1) + fib(n-2)
}

// assert(fib(10) == 55)

let fib_table = #[]
fun memoized_fib(n) {
    if contains(fib_table, n) {
        return fib_table[n]
    }
    
    if n <= 2 {
        fib_table[n] = 1
        return 1
    }
    let answer = memoized_fib(n-1) + memoized_fib(n-2) 
    fib_table[n] = answer
    return answer
}

fun fib_iter(n) {
    let fib = [0, 1]
    for i : 2..n+1 {
        let res = fib[i - 1] + fib[i - 2]
        append(fib, res)
    }
    return fib[len(fib) - 1]
}
