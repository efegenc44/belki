import "math"

fun map(func, list) {
    let i = 0
    while i < len(list) {
        // print(i)
        list[i] = func(list[i])
        i = i + 1
    }
}

fun print_list(list) {
    let i = 0
    while i < len(list) {
        print(list[i])
        i = i + 1
    }
}

fun mul2(n) {
    return n*2
}

fun square(n) {
    return n*n
}

fun pow(base, power) {
    let res = 1;
    let i = 0;
    while i < power {
        res = res * base
        i = i + 1
    }
    return res
}
