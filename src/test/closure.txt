fun closure(a) {
    fun inner() {
        return a
    }
    return inner
}

fun closure2(a) {
    fun inner() {
        fun inner2() {
            return a
        }
        return inner2
    }
    return inner
}

let a = 45
let res = closure(23)()
assert(res != 45)
assert(res == 23)

let b = "wow"
let res2 = closure2("uwu")()()
assert(res2 == "uwu")
assert(!(res2 == "wow"))
