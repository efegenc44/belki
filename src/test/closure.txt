fun closure(a) {
    return fun {
        return a
    }
}

fun closure2(a) {
    return fun {
        return fun {
            return a
        }
    }
}

let a = 45
let res = closure(23)()
assert(res != 45)
assert(res == 23)

let b = "wow"
let res2 = closure2("uwu")()()
assert(res2 == "uwu")
assert(!(res2 == "wow"))
