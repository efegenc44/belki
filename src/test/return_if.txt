fun return_if(n) {
    if n == 3 {
        return 3
        return 4
    } else n == 5 {
        return 5
        return 6
    } else {
        return 7
        return 8
    }
}

assert(return_if(3) == 3)
assert(return_if(3) != 4)
assert(return_if(5) == 5)
assert(return_if(5) != 6)
assert(return_if(31) == 7)
assert(return_if(31) != 8)
